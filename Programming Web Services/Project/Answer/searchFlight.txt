    private List<FlightInfo> searchDirectFlight(String arrivalCity, List<Timetable> timetables, Date departureDate, String departureCity) {
        // Search directly go out flights
        List<FlightInfo> goFlights = new ArrayList<FlightInfo>();
        for (Timetable timetable : timetables) {
            FlightInfo flight = timetable.getFlightCode();
            if (timetable.getDate().equals(departureDate) && flight.getDepartureCity().equals(departureCity) && flight.getArrivalCity().equals(arrivalCity)) {
                goFlights.add(flight);
            }
        }
        return goFlights;
    }

    private List<FlightPair> searchOneStopFlight(List<Timetable> timetables, String departureCity, String arrivalCity, Date departureDate) {
        // Search oneStop go out flights
        List<FlightInfo> flight1s = new ArrayList<FlightInfo>();
        List<FlightInfo> flight2s = new ArrayList<FlightInfo>();
        List<FlightPair> goOneStopFlights = new ArrayList<FlightPair>();
        for (Timetable timetable : timetables) {
            FlightInfo flight = timetable.getFlightCode();
            if (timetable.getDate().equals(departureDate) && flight.getDepartureCity().equals(departureCity)) {
                flight1s.add(flight);
            }
        }

        for (Timetable timetable : timetables) {
            FlightInfo flight = timetable.getFlightCode();
            if (timetable.getDate().equals(departureDate) && flight.getArrivalCity().equals(arrivalCity)) {
                flight2s.add(flight);
            }
        }

        for (FlightInfo flight1 : flight1s) {
            for (FlightInfo flight2 : flight2s) {
                boolean stopCityEqual = flight1.getArrivalCity().equals(flight2.getDepartureCity());
                boolean timeAvailable = flight1.getArrivalTime().compareTo(flight2.getDepartureTime()) < 0;
                if (stopCityEqual && timeAvailable) {
                    goOneStopFlights.add(new FlightPair(flight1, flight2));
                }
            }
        }
        return goOneStopFlights;
    }
